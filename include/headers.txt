+------------------------------------------+
|Conductor:                                |
+------------------------------------------+
|  -controllerHandler: ControllerHandler   |
|  -scheduleHandler: ScheduleHandler       |
|  -physicsHandler: PhysicsHandler         |
|  -drawHandler: DrawHandler               |
|  -audioHandler: AudioHandler             |
+------------------------------------------+
|  +flush()                                |
+------------------------------------------+
                                           

+-------------------------------+
|ControllerHandler:             |
+-------------------------------+
|  -controlables: Controlable[] |
|  -timeHeldDown:               |
+-------------------------------+
|  +flush(): void               |
+-------------------------------+

+--------------------------------------------------+
|ScheduleHandler:                                  |
+--------------------------------------------------+
|  -functions: list<function<int(void)>>[]         |
|  -position: int                                  |
+--------------------------------------------------+
|  +add(f: function<int(void)>, delay: int): void  |
|  +flush(): void                                  |
+--------------------------------------------------+

+------------------------------------------------------+
|Controlable:                                          |
+------------------------------------------------------+
+------------------------------------------------------+
|  +keyHeldEvent(key: Key, timePressed: int): void     |
|  +keyNotHeldEvent(key: Key, timePressed: int): void  |
|  +keyPressedEvent(key: Key, timePressed: int): void  |
|  +keyReleasedEvent(key: Key, timePressed: int): void |
+------------------------------------------------------+




main():
  // window init
  conductor = Conductor()
  while:
    conductor.flush()
    // window update
  // window kill

Conductor:
  flush(){
    controllerHandler.flush()
    scheduleHandler.flush()
    physicsHandler.flush()
    drawHandler.flush()
    audioHandler.flush()
    
}

ControllerHandler:
  flushCurrentFrame(){
    for c in controlables:
      c.flush()
 }
